147     private String readFileHeader(String filename) {
148         String value;
149         try {
150             value = WaarpStringUtils.readFileException(filename);
151         } catch (InvalidArgumentException e) {
152             logger.error("Error while trying to open: " + filename, e);
153             return "";
154         } catch (FileTransferException e) {
155             logger.error("Error while trying to read: " + filename, e);
156             return "";
157         }
158         StringBuilder builder = new StringBuilder(value);
159 
160         WaarpStringUtils.replace(builder, REPLACEMENT.XXXDATEXXX.toString(),
161                 (new Date()).toString());
162         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLOCACTIVEXXX.toString(),
163                 Integer.toString(
164                         Configuration.configuration.getLocalTransaction().
165                                 getNumberLocalChannel()));
166         WaarpStringUtils.replace(builder, REPLACEMENT.XXXNETACTIVEXXX.toString(),
167                 Integer.toString(
168                         Configuration.configuration.getLocalTransaction().
169                                 getNumberLocalChannel()));
170         WaarpStringUtils.replace(builder, REPLACEMENT.XXXHOSTIDXXX.toString(),
171                 Configuration.configuration.getHOST_ID());
172         TrafficCounter trafficCounter =
173                 Configuration.configuration.getGlobalTrafficShapingHandler().trafficCounter();
174         WaarpStringUtils.replace(builder, REPLACEMENT.XXXBANDWIDTHXXX.toString(),
175                 "IN:" + (trafficCounter.lastReadThroughput() / 131072) +
176                         "Mbits&nbsp;&nbsp;OUT:" +
177                         (trafficCounter.lastWriteThroughput() / 131072) + "Mbits");
178         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLANGXXX.toString(), lang);
179         return builder.toString();
180     }

182     private String getTrimValue(String varname) {
183         String value = null;
184         try {
185             value = params.get(varname).get(0).trim();
186         } catch (NullPointerException e) {
187             return null;
188         }
189         if (value.isEmpty()) {
190             value = null;
191         }
192         return value;
193     }

195     @Override
196     protected void channelRead0(ChannelHandlerContext ctx, FullHttpRequest msg) throws Exception {
197         isCurrentRequestXml = false;
198         isCurrentRequestJson = false;
199         status = HttpResponseStatus.OK;
200         FullHttpRequest request = this.request = msg;
201         QueryStringDecoder queryStringDecoder = new QueryStringDecoder(request.uri());
202         uriRequest = queryStringDecoder.path();
203         logger.debug("Msg: " + uriRequest);
204         if (uriRequest.contains("gre/") || uriRequest.contains("img/") ||
205                 uriRequest.contains("res/") || uriRequest.contains("favicon.ico")) {
206             HttpWriteCacheEnable.writeFile(request,
207                     ctx, Configuration.configuration.getHttpBasePath() + uriRequest,
208                     "XYZR66NOSESSION");
209             return;
210         }
211         char cval = 'z';
212         long nb = LIMITROW;
213         // check the URI
214         if (uriRequest.equalsIgnoreCase("/statusxml")) {
215             cval = '5';
216             nb = 0; // since it could be the default or setup by request
217             isCurrentRequestXml = true;
218         } else if (uriRequest.equalsIgnoreCase("/statusjson")) {
219             cval = '7';
220             nb = 0; // since it could be the default or setup by request
221             isCurrentRequestJson = true;
222         }
223         if (request.method() == HttpMethod.GET) {
224             params = queryStringDecoder.parameters();
225         }
226         boolean getMenu = (cval == 'z');
227         boolean extraBoolean = false;
228         if (!params.isEmpty()) {
229             String langarg = getTrimValue("setLng");
230             if (langarg != null && !langarg.isEmpty()) {
231                 lang = langarg;
232             }
233         }
234         if (getMenu) {
235             responseContent.append(REQUEST.index.readFileUnique(this));
236         } else {
237             // Use value 0=Active 1=Error 2=Done 3=All
238             switch (cval) {
239                 case '5':
240                     statusxml(ctx, nb, extraBoolean);
241                     break;
242                 case '7':
243                     statusjson(ctx, nb, extraBoolean);
244                     break;
245                 default:
246                     responseContent.append(REQUEST.index.readFileUnique(this));
247             }
248         }
249         writeResponse(ctx);
250     }

268     /**
269      * Write the response
270      * 
271      */
272     private void writeResponse(ChannelHandlerContext ctx) {
273         // Convert the response content to a ByteBuf.
274         ByteBuf buf = Unpooled.copiedBuffer(responseContent.toString(), WaarpStringUtils.UTF8);
275         responseContent.setLength(0);
276         // Decide whether to close the connection or not.
277         boolean keepAlive = HttpUtil.isKeepAlive(request);
278         boolean close = HttpHeaderValues.CLOSE.contentEqualsIgnoreCase(request
279                 .headers().get(HttpHeaderNames.CONNECTION)) ||
280                 (!keepAlive);
281 
282         // Build the response object.
283         FullHttpResponse response = new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, status, buf);
284         response.headers().add(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());
285         if (isCurrentRequestXml) {
286             response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/xml");
287         } else if (isCurrentRequestJson) {
288             response.headers().set(HttpHeaderNames.CONTENT_TYPE, "application/json");
289         } else {
290             response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
291         }
292         if (keepAlive) {
293             response.headers().set(HttpHeaderNames.CONNECTION,
294                     HttpHeaderValues.KEEP_ALIVE);
295         }
296         if (!close) {
297             // There's no need to add 'Content-Length' header
298             // if this is the last response.
299             response.headers().set(HttpHeaderNames.CONTENT_LENGTH, String.valueOf(buf.readableBytes()));
300         }
301 
302         String cookieString = request.headers().get(HttpHeaderNames.COOKIE);
303         if (cookieString != null) {
304             Set<Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieString);
305             boolean i18nextFound = false;
306             if (!cookies.isEmpty()) {
307                 // Reset the cookies if necessary.
308                 for (Cookie cookie : cookies) {
309                     if (cookie.name().equalsIgnoreCase(I18NEXT)) {
310                         i18nextFound = true;
311                         cookie.setValue(lang);
312                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
313                     } else {
314                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
315                     }
316                 }
317                 if (!i18nextFound) {
318                     Cookie cookie = new DefaultCookie(I18NEXT, lang);
319                     response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
320                 }
321             }
322             if (!i18nextFound) {
323                 Cookie cookie = new DefaultCookie(I18NEXT, lang);
324                 response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
325             }
326         }
327 
328         // Write the response.
329         ChannelFuture future = ctx.writeAndFlush(response);
330         // Close the connection after the write operation is done if necessary.
331         if (close) {
332             future.addListener(ChannelFutureListener.CLOSE);
333         }
334     }

336     /**
337      * Send an error and close
338      * 
339      * @param ctx
340      * @param status
341      */
342     private void sendError(ChannelHandlerContext ctx, HttpResponseStatus status) {
343         responseContent.setLength(0);
344         responseContent.append(REQUEST.error.readHeader(this)).append("OpenR66 Web Failure: ")
345                 .append(status.toString()).append(REQUEST.error.readEnd());
346         ByteBuf buf = Unpooled.copiedBuffer(responseContent.toString(), WaarpStringUtils.UTF8);
347         FullHttpResponse response = new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, status, buf);
348         response.headers().add(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());
349         response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
350         responseContent.setLength(0);
351         // Close the connection as soon as the error message is sent.
352         ctx.writeAndFlush(response).addListener(ChannelFutureListener.CLOSE);
353     }

355     @Override
356     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause)
357             throws Exception {
358         OpenR66Exception exception = OpenR66ExceptionTrappedFactory
359                 .getExceptionFromTrappedException(ctx.channel(), cause);
360         if (exception != null) {
361             if (!(exception instanceof OpenR66ProtocolBusinessNoWriteBackException)) {
362                 if (cause instanceof IOException) {
363                     // Nothing to do
364                     return;
365                 }
366                 logger.warn("Exception in HttpHandler {}", exception.getMessage());
367             }
368             if (ctx.channel().isActive()) {
369                 sendError(ctx, HttpResponseStatus.BAD_REQUEST);
370             }
371         } else {
372             // Nothing to do
373             return;
374         }
375     }

383     @Override
384     public void channelActive(ChannelHandlerContext ctx) throws Exception {
385         logger.debug("Connected");
386         authentHttp.getAuth().specialNoSessionAuth(false, Configuration.configuration.getHOST_ID());
387         super.channelActive(ctx);
388         ChannelGroup group = Configuration.configuration.getHttpChannelGroup();
389         if (group != null) {
390             group.add(ctx.channel());
391         }
