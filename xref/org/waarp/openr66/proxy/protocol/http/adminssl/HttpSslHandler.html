141     private String readFileHeader(String filename) {
142         String value;
143         try {
144             value = WaarpStringUtils.readFileException(filename);
145         } catch (InvalidArgumentException e) {
146             logger.error("Error while trying to open: " + filename, e);
147             return "";
148         } catch (FileTransferException e) {
149             logger.error("Error while trying to read: " + filename, e);
150             return "";
151         }
152         StringBuilder builder = new StringBuilder(value);
153         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLOCALXXX.toString(),
154                 Integer.toString(
155                         Configuration.configuration.getLocalTransaction().
156                                 getNumberLocalChannel()) + " " + Thread.activeCount());
157         WaarpStringUtils.replace(builder, REPLACEMENT.XXXNETWORKXXX.toString(),
158                 Integer.toString(
159                         Configuration.configuration.getLocalTransaction().
160                                 getNumberLocalChannel()));
161         WaarpStringUtils.replace(builder, REPLACEMENT.XXXHOSTIDXXX.toString(),
162                 Configuration.configuration.getHOST_ID());
163         if (authentHttp.isAuthenticated()) {
164             WaarpStringUtils.replace(builder, REPLACEMENT.XXXADMINXXX.toString(),
165                     Messages.getString("HttpSslHandler.1")); //$NON-NLS-1$
166         } else {
167             WaarpStringUtils.replace(builder, REPLACEMENT.XXXADMINXXX.toString(),
168                     Messages.getString("HttpSslHandler.0")); //$NON-NLS-1$
169         }
170         TrafficCounter trafficCounter =
171                 Configuration.configuration.getGlobalTrafficShapingHandler().trafficCounter();
172         WaarpStringUtils.replace(builder, REPLACEMENT.XXXBANDWIDTHXXX.toString(),
173                 Messages.getString("HttpSslHandler.IN") + (trafficCounter.lastReadThroughput() >> 20) + //$NON-NLS-1$
174                 Messages.getString("HttpSslHandler.MOPS") + //$NON-NLS-1$
175                 Messages.getString("HttpSslHandler.OUT") + //$NON-NLS-1$
176                 (trafficCounter.lastWriteThroughput() >> 20) +
177                 Messages.getString("HttpSslHandler.MOPS")); //$NON-NLS-1$
178         WaarpStringUtils.replace(builder, REPLACEMENT.XXXLANGXXX.toString(), lang);
179         return builder.toString();
180     }

281     private String System() {
282         getParams();
283         if (params == null) {
284             String system = REQUEST.System.readFileUnique(this);
285             StringBuilder builder = new StringBuilder(system);
286             replaceStringSystem(builder);
287             langHandle(builder);
288             return builder.toString();
289         }
290         String extraInformation = null;
291         if (params.containsKey("ACTION")) {
292             List<String> action = params.get("ACTION");
293             for (String act : action) {
294                 if (act.equalsIgnoreCase("Language")) {
295                     lang = getTrimValue("change");
296                     String sys = getTrimValue("changesys");
297                     Messages.init(new Locale(sys));
298                     extraInformation = Messages.getString("HttpSslHandler.LangIs") + "Web: " + lang + " OpenR66: " + Messages.getSlocale(); //$NON-NLS-1$
299                 } else if (act.equalsIgnoreCase("Level")) {
300                     String loglevel = getTrimValue("loglevel");
301                     WaarpLogLevel level = WaarpLogLevel.WARN;
302                     if (loglevel.equalsIgnoreCase("debug")) {
303                         level = WaarpLogLevel.DEBUG;
304                     } else if (loglevel.equalsIgnoreCase("info")) {
305                         level = WaarpLogLevel.INFO;
306                     } else if (loglevel.equalsIgnoreCase("warn")) {
307                         level = WaarpLogLevel.WARN;
308                     } else if (loglevel.equalsIgnoreCase("error")) {
309                         level = WaarpLogLevel.ERROR;
310                     }
311                     WaarpLoggerFactory.setLogLevel(level);
312                     extraInformation = Messages.getString("HttpSslHandler.LangIs") + level.name(); //$NON-NLS-1$
313                 } else if (act.equalsIgnoreCase("Disconnect")) {
314                     String logon = Logon();
315                     logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
316                             Messages.getString("HttpSslHandler.DisActive"));
317                     newSession = true;
318                     clearSession();
319                     forceClose = true;
320                     return logon;
321                 } else if (act.equalsIgnoreCase("Shutdown")) {
322                     String error;
323                     if (Configuration.configuration.getShutdownConfiguration().serviceFuture != null) {
324                         error = error(Messages.getString("HttpSslHandler.38")); //$NON-NLS-1$
325                     } else {
326                         error = error(Messages.getString("HttpSslHandler.37")); //$NON-NLS-1$
327                     }
328                     R66ShutdownHook.setRestart(false);
329                     newSession = true;
330                     clearSession();
331                     forceClose = true;
332                     shutdown = true;
333                     return error;
334                 } else if (act.equalsIgnoreCase("Restart")) {
335                     String error;
336                     if (Configuration.configuration.getShutdownConfiguration().serviceFuture != null) {
337                         error = error(Messages.getString("HttpSslHandler.38")); //$NON-NLS-1$
338                     } else {
339                         error = error(Messages.getString("HttpSslHandler.39") + (Configuration.configuration.getTIMEOUTCON() * 2 / 1000) + Messages.getString("HttpSslHandler.40")); //$NON-NLS-1$ //$NON-NLS-2$
340                     }
341                     error = error.replace("XXXRELOADHTTPXXX", "HTTP-EQUIV=\"refresh\" CONTENT=\""
342                             + (Configuration.configuration.getTIMEOUTCON() * 2 / 1000) + "\"");
343                     R66ShutdownHook.setRestart(true);
344                     newSession = true;
345                     clearSession();
346                     forceClose = true;
347                     shutdown = true;
348                     return error;
349                 } else if (act.equalsIgnoreCase("Validate")) {
350                     String bsessionr = getTrimValue("BSESSR");
351                     long lsessionr = Configuration.configuration.getServerChannelReadLimit();
352                     long lglobalr;
353                     long lsessionw;
354                     long lglobalw;
355                     try {
356                         if (bsessionr != null) {
357                             lsessionr = (Long.parseLong(bsessionr) / 10) * 10;
358                         }
359                         String bglobalr = getTrimValue("BGLOBR");
360                         lglobalr = Configuration.configuration.getServerGlobalReadLimit();
361                         if (bglobalr != null) {
362                             lglobalr = (Long.parseLong(bglobalr) / 10) * 10;
363                         }
364                         String bsessionw = getTrimValue("BSESSW");
365                         lsessionw = Configuration.configuration.getServerChannelWriteLimit();
366                         if (bsessionw != null) {
367                             lsessionw = (Long.parseLong(bsessionw) / 10) * 10;
368                         }
369                         String bglobalw = getTrimValue("BGLOBW");
370                         lglobalw = Configuration.configuration.getServerGlobalWriteLimit();
371                         if (bglobalw != null) {
372                             lglobalw = (Long.parseLong(bglobalw) / 10) * 10;
373                         }
374                         Configuration.configuration.changeNetworkLimit(
375                                 lglobalw, lglobalr, lsessionw, lsessionr,
376                                 Configuration.configuration.getDelayLimit());
377                         String dcomm = getTrimValue("DCOM");
378                         if (dcomm != null) {
379                             Configuration.configuration.setDelayCommander(Long.parseLong(dcomm));
380                             if (Configuration.configuration.getDelayCommander() <= 100) {
381                                 Configuration.configuration.setDelayCommander(100);
382                             }
383                             Configuration.configuration.reloadCommanderDelay();
384                         }
385                         String dret = getTrimValue("DRET");
386                         if (dret != null) {
387                             Configuration.configuration.setDelayRetry(Long.parseLong(dret));
388                             if (Configuration.configuration.getDelayRetry() <= 1000) {
389                                 Configuration.configuration.setDelayRetry(1000);
390                             }
391                         }
392                         extraInformation = Messages.getString("HttpSslHandler.41"); //$NON-NLS-1$
393                     } catch (NumberFormatException e) {
394                         extraInformation = Messages.getString("HttpSslHandler.42"); //$NON-NLS-1$
395                     }
396                 }
397             }
398         }
399         String system = REQUEST.System.readFileUnique(this);
400         StringBuilder builder = new StringBuilder(system);
401         replaceStringSystem(builder);
402         langHandle(builder);
403         if (extraInformation != null) {
404             builder.append(extraInformation);
405         }
406         return builder.toString();
407     }

409     private void getParams() {

411             params = null;
412         } else if (request.method() == HttpMethod.POST) {
413             ByteBuf content = request.content();
414             if (content.isReadable()) {
415                 String param = content.toString(WaarpStringUtils.UTF8);
416                 QueryStringDecoder queryStringDecoder2 = new QueryStringDecoder("/?" + param);
417                 params = queryStringDecoder2.parameters();
418             } else {
419                 params = null;
420             }
421         }
422     }

435     private void checkAuthent(ChannelHandlerContext ctx) {
436         newSession = true;
437         if (request.method() == HttpMethod.GET) {
438             String logon = Logon();
439             logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
440                     "");
441             responseContent.append(logon);
442             clearSession();
443             writeResponse(ctx);
444             return;
445         } else if (request.method() == HttpMethod.POST) {
446             getParams();
447             if (params == null) {
448                 String logon = Logon();
449                 logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
450                         Messages.getString("HttpSslHandler.EmptyLogin"));
451                 responseContent.append(logon);
452                 clearSession();
453                 writeResponse(ctx);
454                 return;
455             }
456         }
457         boolean getMenu = false;
458         if (params.containsKey("Logon")) {
459             String name = null, password = null;
460             List<String> values = null;
461             if (!params.isEmpty()) {
462                 // get values
463                 if (params.containsKey("name")) {
464                     values = params.get("name");
465                     if (values != null) {
466                         name = values.get(0);
467                         if (name == null || name.isEmpty()) {
468                             getMenu = true;
469                         }
470                     }
471                 } else {
472                     getMenu = true;
473                 }
474                 // search the nb param
475                 if ((!getMenu) && params.containsKey("passwd")) {
476                     values = params.get("passwd");
477                     if (values != null) {
478                         password = values.get(0);
479                         if (password == null || password.isEmpty()) {
480                             getMenu = true;
481                         } else {
482                             getMenu = false;
483                         }
484                     } else {
485                         getMenu = true;
486                     }
487                 } else {
488                     getMenu = true;
489                 }
490             } else {
491                 getMenu = true;
492             }
493             if (!getMenu) {
494                 logger.debug("Name=" + name + " vs "
495                         + name.equals(Configuration.configuration.getADMINNAME()) +
496                         " Passwd vs " + Arrays.equals(password.getBytes(WaarpStringUtils.UTF8),
497                                 Configuration.configuration.getSERVERADMINKEY()));
498                 if (name.equals(Configuration.configuration.getADMINNAME()) &&
499                         Arrays.equals(password.getBytes(WaarpStringUtils.UTF8),
500                                 Configuration.configuration.getSERVERADMINKEY())) {
501                     authentHttp.getAuth().specialNoSessionAuth(true,
502                             Configuration.configuration.getHOST_ID());
503                     authentHttp.setStatus(70);
504                 } else {
505                     getMenu = true;
506                 }
507                 if (!authentHttp.isAuthenticated()) {
508                     authentHttp.setStatus(71);
509                     logger.debug("Still not authenticated: {}", authentHttp);
510                     getMenu = true;
511                 }
512             }
513         } else {
514             getMenu = true;
515         }
516         if (getMenu) {
517             String logon = Logon();
518             logon = logon.replaceAll(REPLACEMENT.XXXERRORMESGXXX.toString(),
519                     Messages.getString("HttpSslHandler.BadLogin"));
520             responseContent.append(logon);
521             clearSession();
522             writeResponse(ctx);
523         } else {
524             String index = index();
525             responseContent.append(index);
526             clearSession();
527             admin = new DefaultCookie(R66SESSION + Configuration.configuration.getHOST_ID(),
528                     Configuration.configuration.getHOST_ID() +
529                             Long.toHexString(random.nextLong()));
530             sessions.put(admin.value(), this.authentHttp);
531             authentHttp.setStatus(72);
532             logger.debug("CreateSession: " + uriRequest + ":{}", admin);
533             writeResponse(ctx);
534         }
535     }

537     @Override
538     protected void channelRead0(ChannelHandlerContext ctx, FullHttpRequest msg) throws Exception {
539         FullHttpRequest request = this.request = msg;
540         QueryStringDecoder queryStringDecoder = new QueryStringDecoder(request.uri());
541         uriRequest = queryStringDecoder.path();
542         logger.debug("Msg: " + uriRequest);
543         if (uriRequest.contains("gre/") || uriRequest.contains("img/") ||
544                 uriRequest.contains("res/") || uriRequest.contains("favicon.ico")) {
545             HttpWriteCacheEnable.writeFile(request,
546                     ctx, Configuration.configuration.getHttpBasePath() + uriRequest,
547                     R66SESSION + Configuration.configuration.getHOST_ID());
548             ctx.flush();
549             return;
550         }
551         checkSession(ctx.channel());
552         if (!authentHttp.isAuthenticated()) {
553             logger.debug("Not Authent: " + uriRequest + ":{}", authentHttp);
554             checkAuthent(ctx);
555             return;
556         }
557         String find = uriRequest;
558         if (uriRequest.charAt(0) == '/') {
559             find = uriRequest.substring(1);
560         }
561         find = find.substring(0, find.indexOf("."));
562         REQUEST req = REQUEST.index;
563         try {
564             req = REQUEST.valueOf(find);
565         } catch (IllegalArgumentException e1) {
566             req = REQUEST.index;
567             logger.debug("NotFound: " + find + ":" + uriRequest);
568         }
569         switch (req) {
570             case index:
571                 responseContent.append(index());
572                 break;
573             case Logon:
574                 responseContent.append(index());
575                 break;
576             case System:
577                 responseContent.append(System());
578                 break;
579             default:
580                 responseContent.append(index());
581                 break;
582         }
583         writeResponse(ctx);
584     }

586     private void checkSession(Channel channel) {
587         String cookieString = request.headers().get(HttpHeaderNames.COOKIE);
588         if (cookieString != null) {
589             Set<Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieString);
590             if (!cookies.isEmpty()) {
591                 for (Cookie elt : cookies) {
592                     if (elt.name().equalsIgnoreCase(R66SESSION + Configuration.configuration.getHOST_ID())) {
593                         logger.debug("Found session: " + elt);
594                         admin = elt;
595                         R66Session session = sessions.get(admin.value());
596                         if (session != null) {
597                             authentHttp = session;
598                             authentHttp.setStatus(73);
599                         } else {
600                             admin = null;
601                             continue;
602                         }
603                     } else if (elt.name().equalsIgnoreCase(I18NEXT)) {
604                         logger.debug("Found i18next: " + elt);
605                         lang = elt.value();
606                     }
607                 }
608             }
609         }
610         if (admin == null) {
611             logger.debug("NoSession: " + uriRequest + ":{}", admin);
612         }
613     }
614 
615     private void handleCookies(HttpResponse response) {
616         String cookieString = request.headers().get(HttpHeaderNames.COOKIE);
617         boolean i18nextFound = false;
618         if (cookieString != null) {
619             Set<Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieString);
620             if (!cookies.isEmpty()) {
621                 // Reset the sessions if necessary.
622                 boolean findSession = false;
623                 for (Cookie cookie : cookies) {
624                     if (cookie.name().equalsIgnoreCase(R66SESSION + Configuration.configuration.getHOST_ID())) {
625                         if (newSession) {
626                             findSession = false;
627                         } else {
628                             findSession = true;
629                             response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
630                         }
631                     } else if (cookie.name().equalsIgnoreCase(I18NEXT)) {
632                         i18nextFound = true;
633                         cookie.setValue(lang);
634                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
635                     } else {
636                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
637                     }
638                 }
639                 if (!i18nextFound) {
640                     Cookie cookie = new DefaultCookie(I18NEXT, lang);
641                     response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
642                 }
643                 newSession = false;
644                 if (!findSession) {
645                     if (admin != null) {
646                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(admin));
647                         logger.debug("AddSession: " + uriRequest + ":{}", admin);
648                     }
649                 }
650             }
651         } else {
652             Cookie cookie = new DefaultCookie(I18NEXT, lang);
653             response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
654             if (admin != null) {
655                 logger.debug("AddSession: " + uriRequest + ":{}", admin);
656                 response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(admin));
657             }
658         }
659     }

661     /**
662      * Write the response
663      * 
664      */
665     private void writeResponse(ChannelHandlerContext ctx) {
666         // Convert the response content to a ByteBuf.
667         ByteBuf buf = Unpooled.copiedBuffer(responseContent.toString(),
668                 WaarpStringUtils.UTF8);
669         responseContent.setLength(0);
670 
671         // Decide whether to close the connection or not.
672         boolean keepAlive = HttpUtil.isKeepAlive(request);
673         boolean close = HttpHeaderValues.CLOSE.contentEqualsIgnoreCase(request
674                 .headers().get(HttpHeaderNames.CONNECTION)) ||
675                 (!keepAlive) || forceClose;
676 
677         // Build the response object.
678         FullHttpResponse response = new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, HttpResponseStatus.OK, buf);
679         response.headers().add(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());
680         response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
681         if (keepAlive) {
682             response.headers().set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
683         }
684         if (!close) {
685             // There's no need to add 'Content-Length' header
686             // if this is the last response.
687             response.headers().set(HttpHeaderNames.CONTENT_LENGTH,
688                     String.valueOf(buf.readableBytes()));
689         }
690 
691         handleCookies(response);
692 
693         // Write the response.
694         ChannelFuture future = ctx.writeAndFlush(response);
695         // Close the connection after the write operation is done if necessary.
696         if (close) {
697             future.addListener(WaarpSslUtility.SSLCLOSE);
698         }
699         if (shutdown) {
700             ChannelUtils.startShutdown();
701         }
702     }

704     /**
705      * Send an error and close
706      * 
707      * @param ctx
708      * @param status
709      */
710     private void sendError(ChannelHandlerContext ctx, HttpResponseStatus status) {
711         responseContent.setLength(0);
712         responseContent.append(error(status.toString()));
713         FullHttpResponse response = new DefaultFullHttpResponse(
714                 HttpVersion.HTTP_1_1, status,
715                 Unpooled.copiedBuffer(responseContent.toString(),
716                         WaarpStringUtils.UTF8));
717         response.headers().add(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());
718         response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
719         responseContent.setLength(0);
720         clearSession();
721         // Close the connection as soon as the error message is sent.
722         ctx.writeAndFlush(response).addListener(WaarpSslUtility.SSLCLOSE);
723     }

725     @Override
726     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause)
727             throws Exception {
728         OpenR66Exception exception = OpenR66ExceptionTrappedFactory
729                 .getExceptionFromTrappedException(ctx.channel(), cause);
730         if (exception != null) {
731             if (!(exception instanceof OpenR66ProtocolBusinessNoWriteBackException)) {
732                 if (cause instanceof IOException) {
733                     // Nothing to do
734                     return;
735                 }
736                 logger.warn("Exception in HttpSslHandler {}", exception.getMessage());
737             }
738             if (ctx.channel().isActive()) {
739                 sendError(ctx, HttpResponseStatus.BAD_REQUEST);
740             }
741         } else {
742             // Nothing to do
743             return;
744         }
745     }
