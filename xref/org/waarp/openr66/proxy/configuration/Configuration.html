45  public class Configuration extends org.waarp.openr66.protocol.configuration.Configuration {

53  	 * 
54  	 */
55      public Configuration() {
56          super();
57      }

59      @Override
60      public void computeNbThreads() {
61          int nb = Runtime.getRuntime().availableProcessors() + 1;
62          SERVER_THREAD = nb;
63          CLIENT_THREAD = SERVER_THREAD + 1;
64          RUNNER_THREAD = 10;
65      }
66  
67      @Override
68      public void serverStartup() {
69          isServer = true;
70          shutdownConfiguration.timeout = TIMEOUTCON;
71          R66ShutdownHook.addShutdownHook();
72          if ((!useNOSSL) && (!useSSL)) {
73              logger.error("OpenR66 has neither NOSSL nor SSL support included! Stop here!");
74              System.exit(-1);
75          }
76          pipelineInit();
77          serverPipelineInit();
78          r66Startup();
79          startHttpSupport();
80          try {
81              startMonitoring();
82          } catch (WaarpDatabaseSqlException e) {
83          }
84          logger.warn(this.toString());
85      }
86  
87      @Override
88      public void r66Startup() {
89          logger.debug("Start R66: " + SERVER_PORT + ":" + useNOSSL + " " + SERVER_SSLPORT + ":"
90                  + useSSL + ":" + HOST_SSLID);
91          // add into configuration
92          this.constraintLimitHandler.setServer(true);
93          // Global Server
94          serverChannelGroup = new DefaultChannelGroup("OpenR66", subTaskGroup.next());
95          if (useNOSSL) {
96              serverBootstrap = new ServerBootstrap();
97              WaarpNettyUtil.setServerBootstrap(serverBootstrap, bossGroup, workerGroup, (int) TIMEOUTCON);
98              networkServerInitializer = new NetworkServerInitializer(true);
99              serverBootstrap.childHandler(networkServerInitializer);
100             // FIXME take into account multiple address
101             for (ProxyEntry entry : ProxyEntry.proxyEntries.values()) {
102                 if (!entry.isLocalSsl()) {
103                     ChannelFuture future = serverBootstrap.bind(entry.getLocalSocketAddress()).awaitUninterruptibly();
104                     if (future.isSuccess()) {
105                         bindNoSSL = future.channel();
106                         serverChannelGroup.add(bindNoSSL);
107                     } else {
108                         logger.warn(Messages.getString("Configuration.NOSSLDeactivated")
109                                 + " for " + entry.getLocalSocketAddress()); //$NON-NLS-1$
110                     }
111                 }
112             }
113         } else {
114             networkServerInitializer = null;
115             logger.warn(Messages.getString("Configuration.NOSSLDeactivated")); //$NON-NLS-1$
116         }
117 
118         if (useSSL && HOST_SSLID != null) {
119             serverSslBootstrap = new ServerBootstrap();
120             WaarpNettyUtil.setServerBootstrap(serverSslBootstrap, bossGroup, workerGroup, (int) TIMEOUTCON);
121             networkSslServerInitializer = new NetworkSslServerInitializer(false);
122             serverSslBootstrap.childHandler(networkSslServerInitializer);
123             // FIXME take into account multiple address
124             for (ProxyEntry entry : ProxyEntry.proxyEntries.values()) {
125                 if (entry.isLocalSsl()) {
126                     ChannelFuture future = serverSslBootstrap.bind(entry.getLocalSocketAddress())
127                             .awaitUninterruptibly();
128                     if (future.isSuccess()) {
129                         bindSSL = future.channel();
130                         serverChannelGroup.add(bindSSL);
131                     } else {
132                         logger.warn(Messages.getString("Configuration.SSLMODEDeactivated")
133                                 + " for " + entry.getLocalSocketAddress()); //$NON-NLS-1$
134                     }
135                 }
136             }
137         } else {
138             networkSslServerInitializer = null;
139             logger.warn(Messages.getString("Configuration.SSLMODEDeactivated")); //$NON-NLS-1$
140         }
141 
142         // Factory for TrafficShapingHandler
143         globalTrafficShapingHandler = new GlobalTrafficHandler(subTaskGroup, serverGlobalWriteLimit,
144                 serverGlobalReadLimit, serverChannelWriteLimit, serverChannelReadLimit, delayLimit);
145         this.constraintLimitHandler.setHandler(globalTrafficShapingHandler);
146         ProxyBridge.initialize();
147         localTransaction = new LocalTransaction();
148         thriftService = null;
149     }
150 
151     @Override
152     public void startHttpSupport() {
153         // Now start the HTTP support
154         logger.info(Messages.getString("Configuration.HTTPStart") + SERVER_HTTPPORT + //$NON-NLS-1$
155                 " HTTPS: " + SERVER_HTTPSPORT);
156         httpChannelGroup = new DefaultChannelGroup("HttpOpenR66", subTaskGroup.next());
157         // Configure the server.
158         httpBootstrap = new ServerBootstrap();
159         WaarpNettyUtil.setServerBootstrap(httpBootstrap, httpBossGroup, httpWorkerGroup, (int) TIMEOUTCON);
160         // Set up the event pipeline factory.
161         httpBootstrap.childHandler(new HttpInitializer(useHttpCompression));
162         // Bind and start to accept incoming connections.
163         if (SERVER_HTTPPORT > 0) {
164             ChannelFuture future = httpBootstrap.bind(new InetSocketAddress(SERVER_HTTPPORT)).awaitUninterruptibly();
165             if (future.isSuccess()) {
166                 httpChannelGroup.add(future.channel());
167             }
168         }
169         // Now start the HTTPS support
170         // Configure the server.
171         httpsBootstrap = new ServerBootstrap();
172         // Set up the event pipeline factory.
173         WaarpNettyUtil.setServerBootstrap(httpsBootstrap, httpBossGroup, httpWorkerGroup, (int) TIMEOUTCON);
174         httpsBootstrap.childHandler(new HttpSslInitializer(useHttpCompression, false));
175         // Bind and start to accept incoming connections.
176         if (SERVER_HTTPSPORT > 0) {
177             ChannelFuture future = httpsBootstrap.bind(new InetSocketAddress(SERVER_HTTPSPORT)).awaitUninterruptibly();
178             if (future.isSuccess()) {
179                 httpChannelGroup.add(future.channel());
180             }
181         }
182     }
183 
184     @Override
185     public void serverStop() {
186         super.serverStop();
187         ProxyBridge.transaction.closeAll();
188     }
